# SMART GRID LOAD DECISION AGENT
# ---------------------------------
# User Input Version using A* Algorithm


import heapq


# ---------------------------------
# Building Class
# ---------------------------------
class Building:

    def init(self, name, load_required, priority, distance):
        self.name = name
        self.load_required = load_required
        self.priority = priority
        self.distance = distance

    def heuristic(self):

        # Heuristic Function h(n)
        return self.distance - self.priority


# ---------------------------------
# Smart Grid Agent
# ---------------------------------
class SmartGridAgent:

    def init(self, buildings, total_power):
        self.buildings = buildings
        self.total_power = total_power

    def allocate_power(self):

        priority_queue = []

        # ---------------------------------
        # A* ALGORITHM USED HERE
        #
        # f(n) = g(n) + h(n)
        #
        # g(n) = actual electricity load
        # h(n) = heuristic value
        #
        # Lower f(n) = better building
        # to supply power first
        # ---------------------------------
        for building in self.buildings:

            g_n = building.load_required
            h_n = building.heuristic()

            f_n = g_n + h_n

            heapq.heappush(priority_queue, (f_n, building))

        print("\nSMART GRID POWER ALLOCATION\n")

        remaining_power = self.total_power

        while priority_queue and remaining_power > 0:

            cost, building = heapq.heappop(priority_queue)

            if remaining_power >= building.load_required:

                print(f"Power Supplied to: {building.name}")
                print(f"Load Used: {building.load_required} units")
                print(f"Priority: {building.priority}")
                print(f"Distance: {building.distance}")
                print("-----------------------------------")

                remaining_power -= building.load_required

            else:

                print(f"Not enough power for: {building.name}")
                print("-----------------------------------")

        print(f"\nRemaining Power: {remaining_power} units")


# ---------------------------------
# USER INPUT SECTION
# ---------------------------------

buildings = []

num_buildings = int(input("Enter number of buildings: "))

for i in range(num_buildings):

    print(f"\nEnter details for Building {i + 1}")

    name = input("Building Name: ")

    load_required = int(input("Electricity Load Required: "))

    priority = int(input("Priority (Higher = More Important): "))

    distance = int(input("Distance from Power Source: "))

    building = Building(name, load_required, priority, distance)

    buildings.append(building)


total_power = int(input("\nEnter Total Available Power: "))


# ---------------------------------
# RUN AGENT
# ---------------------------------

agent = SmartGridAgent(buildings, total_power)

agent.allocate_power()
